import github.com/prashantv/gostub即可使用

//1. stub variable
var someVar = "product value"
stb := gostub.Stub(&someVar, "test value")
defer stb.Reset()

//2. stub 有返回值的func或者无返回值的func
var someFunc = func() string{
    return "product"
}
stb := gostub.Stub(&someFunc, func() string{
    retrun "test"
})

//3. stub 有返回值的func
stb := gostub.StubFunc(&someFunc, "test")

//4. stub 有多个返回值的func
stb := gostub.StubFunc(&someFunc, "test", nil)


//5. stub env
stubs := gostub.New()
stubs.SetEnv("GOSTUB_VAR", "test_value")
defer stubs.Reset()


//6. multiple stub
stubs := gostub.Stub(&v1, 1)
stubs.Stub(&v2, 2)
defer stubs.Reset()

//7. 精简stub
defer gostub.Stub(&v1, 1).Stub(&v2, 2).Reset()
